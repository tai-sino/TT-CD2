    B. Tao 3 service files:

    1. frontend/src/services/sinhVienService.js:
    ```javascript
    import api from './api';

    export function getStudents(params) {
      return api.get('/students', { params }).then(r => r.data);
    }

    export function importStudents(file, kyLvtnId) {
      const formData = new FormData();
      formData.append('file', file);
      formData.append('ky_lvtn_id', kyLvtnId);
      return api.post('/students/import', formData, {
        headers: { 'Content-Type': undefined }
      }).then(r => r.data);
    }

    export function createStudent(data) {
      return api.post('/students', data).then(r => r.data);
    }

    export function updateStudent(mssv, data) {
      return api.put(`/students/${mssv}`, data).then(r => r.data);
    }

    export function deleteStudent(mssv) {
      return api.delete(`/students/${mssv}`).then(r => r.data);
    }
    ```

    QUAN TRONG cho importStudents: set `headers: { 'Content-Type': undefined }` de axios xoa header Content-Type mac dinh va tu set multipart/form-data voi boundary. Neu khong lam dieu nay, api.js da set Content-Type: application/json se override va server khong parse duoc file.
